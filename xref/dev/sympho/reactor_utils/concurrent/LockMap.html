3   import org.checkerframework.checker.nullness.qual.NonNull;
4   import org.checkerframework.checker.nullness.qual.Nullable;
5   
6   /** 
7    * Locking manager that provides distinct locks based on a key value.
8    * 
9    * <p>Unlike a normal lock, the locks provided by this type are not bound to a particular
10   * thread, and so any thread with a reference to the returned lock may release it.
11   *
12   * @param <K> The key type.
13   * @version 1.0
14   * @since 1.0
15   * @apiNote The goal of this type is to provide a relatively low-overhead protection mechanism
16   *          for regions that should be mutually exclusive based on some parameter (rather than
17   *          globaly), and the value space of that parameter is large and/or not known ahead
18   *          of time, making a simple map of normal locks impractical or inneficient.
19   */
20  public interface LockMap<K extends @NonNull Object> {
21  
22      /**
23       * Acquires the lock of the given key, if currently available.
24       *
25       * @param key The key to acquire a lock for.
26       * @return The acquired lock, or {@code null} if the lock is in use and could not be acquired.
27       */
28      @Nullable AcquiredLock tryAcquire( K key );
29  
30      /**
31       * Retrieves a lock in this map.
32       *
33       * @param key The lock's key.
34       * @return The lock with the given key.
35       * @apiNote Using the returned lock is functionally the same as directly calling the
36       *          correspoding methods in the map with the given key. This is mostly a convenience 
37       *          for re-acquiring a lock and for compatibility with APIs that need a lock.
38       * @implSpec Implementations should have very little overhead (in terms of both runtime and 
39       *           memory usage) compared to using the map directly.
40       */
41      Lock get( K key );
42      
43  }
