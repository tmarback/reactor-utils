1   package dev.sympho.reactor_utils.concurrent;
2   
3   import reactor.core.publisher.Flux;
4   import reactor.core.publisher.Mono;
5   
6   /**
7    * A lock that has been acquired.
8    * 
9    * <p>Unlike a normal lock, the code that has a reference to an instance already owns 
10   * the corresponding lock and can only release it. Once released, the lock may not 
11   * be re-acquired through the same instance, instead a new instance must be obtained 
12   * from the original source.
13   * 
14   * <p>Releasing a lock (using any of the provided methods) is a <i>idempotent</i>
15   * operation; it is released on the first call, and any subsequent calls (even to
16   * different methods) have no effect.
17   *
18   * @version 1.0
19   * @since 1.0
20   * @apiNote While binding the ownership of a lock to an instance instead of a thread
21   *          arguably makes it harder to achieve concurrency safety (due to it still
22   *          being possible to make concurrent accesses after acquiring the lock, as
23   *          it is not bound to a particular thread), it is necessary in the context
24   *          of reactive streams where a single chain of operations may execute in
25   *          different threads.
26   */
27  public interface AcquiredLock extends AutoCloseable {
28  
29      /**
30       * Releases the held lock, making it available to be acquired again.
31       */
32      void release();
33  
34      /**
35       * Composes the release of this lock with the given Mono so that the lock is
36       * released after the operation completes.
37       * 
38       * <p>Note that the release occurs <i>after</i> the completion is propagated
39       * downstream. Do not assume that the lock is already available when the
40       * returned Mono terminates.
41       *
42       * @param <T> The data type.
43       * @param source The data source.
44       * @return A Mono that relays the data emitted by the source, and releases
45       *         the lock once it terminates for any reason (including cancellation).
46       */
47      default <T> Mono<T> releaseAfter( final Mono<T> source ) {
48  
49          return source.doFinally( s -> release() );
50  
51      }
52  
53      /**
54       * Composes the release of this lock with the given Flux so that the lock is
55       * released after the operation completes.
56       * 
57       * <p>Note that the release occurs <i>after</i> the completion is propagated
58       * downstream. Do not assume that the lock is already available when the
59       * returned Mono terminates.
60       *
61       * @param <T> The data type.
62       * @param source The data source.
63       * @return A Flux that relays the data emitted by the source, and releases
64       *         the lock once it terminates for any reason (including cancellation).
65       */
66      default <T> Flux<T> releaseAfter( final Flux<T> source ) {
67  
68          return source.doFinally( s -> release() );
69  
70      }
71  
72      /**
73       * Releases the lock.
74       *
75       * @see #release()
76       */
77      @Override
78      default void close() {
79  
80          release();
81  
82      }
83  
84  }
