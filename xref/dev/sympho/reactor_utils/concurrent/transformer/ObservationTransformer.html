1   package dev.sympho.reactor_utils.concurrent.transformer;
2   
3   import java.util.function.Function;
4   
5   import org.checkerframework.checker.nullness.qual.NonNull;
6   import org.checkerframework.checker.nullness.qual.Nullable;
7   import org.checkerframework.dataflow.qual.SideEffectFree;
8   
9   import dev.sympho.reactor_utils.concurrent.AcquiredLock;
10  import io.micrometer.observation.ObservationRegistry;
11  import reactor.core.observability.micrometer.Micrometer;
12  import reactor.core.publisher.Mono;
13  
14  /**
15   * A transformer that creates observations when waiting for a resource.
16   * 
17   * <p>Note that this class requires micrometer-observation to be in the classpath.
18   *
19   * @version 1.0
20   * @since 1.0
21   */
22  public class ObservationTransformer implements LockTransformer, LockMapTransformer<String>, 
23          LatchTransformer {
24  
25      /** The registry to use. */
26      private final ObservationRegistry registry;
27  
28      /** The name that identifies the resource. */
29      private final String name;
30  
31      /**
32       * Creates a new instance.
33       *
34       * @param registry The registry to use.
35       * @param name The name that identifies the resource. Added as a tag in created observations.
36       */
37      public ObservationTransformer( final ObservationRegistry registry, final String name ) {
38  
39          this.registry = registry;
40          this.name = name;
41  
42      }
43  
44      /**
45       * Adds a high cardinality tag to the current observation (if any).
46       *
47       * @param key The tag key.
48       * @param value The tag value.
49       */
50      private void addHighCardinalityTag( final String key, final String value ) {
51  
52          final var observation = registry.getCurrentObservation();
53          if ( observation != null ) {
54              observation.highCardinalityKeyValue( key, value );
55          }
56  
57      }
58  
59      /**
60       * Creates a function that adds a high cardinality tag to the current observation 
61       * (if any) on subscription.
62       *
63       * @param <T> The mono value type.
64       * @param key The tag key.
65       * @param value The tag value. If {@code null}, it does not add the tag.
66       * @return The created function.
67       */
68      private <T extends @NonNull Object> Function<Mono<T>, Mono<T>> highCardinalityTag( 
69              final String key, final @Nullable String value ) {
70  
71          if ( value == null ) {
72              return Function.identity();
73          }
74  
75          return mono -> mono.doOnSubscribe( s -> addHighCardinalityTag( key, value ) );
76  
77      }
78  
79      @Override
80      public Mono<AcquiredLock> transformAcquire( final Mono<AcquiredLock> pending ) {
81  
82          return pending.name( "lock.acquire" )
83                  .tag( "lock.name", name )
84                  .tap( Micrometer.observation( registry ) );
85  
86      }
87  
88      /**
89       * @param key The lock key. If non-{@code null}, it is added to the observation as a
90       *            high-cardinality label.
91       * @see #keyMapped(Function)
92       * @see #noKey()
93       * @implNote As lock maps cannot have {@code null} keys, generally the key will <i>always</i>
94       *           be added as a label. If it is not useful or too high cardinality, use
95       *           {@link #keyMapped(Function)} to reduce the label space or {@link #noKey()} to
96       *           prevent it from being added, or remove the {@code lockmap.key} label using 
97       *           a filter.
98       */
99      @Override
100     @SuppressWarnings( "argument" ) // Weird inference
101     public Mono<AcquiredLock> transformAcquire( 
102             final @Nullable String key, final Mono<AcquiredLock> pending ) {
103 
104         return pending.name( "lockmap.acquire" )
105                 .tag( "lockmap.name", name )
106                 .transform( highCardinalityTag( "lockmap.key", key ) )
107                 .tap( Micrometer.observation( registry ) );
108 
109     }
110 
111     /**
112      * Derives a lock map transformer that uses the given mapper function to convert
113      * keys into strings before delegating to this transformer.
114      *
115      * @param <K> The key type.
116      * @param keyMapper The function to use to transform keys. It may return {@code null}, in
117      *                  which case the key is not used as a label.
118      * @return The transformer.
119      * @see #transformAcquire(String, Mono)
120      * @apiNote The primary purpose of this method is having a way to add observability to lock
121      *          maps with non-string keys; however, it may also be used with string-keyed maps
122      *          in order to apply arbitrary transformations to the key before using it as a label
123      *          (for example, filtering out certain key values or mapping multiple keys to one
124      *          label to reduce the cardinality).
125      */
126     @SideEffectFree
127     public <K extends @NonNull Object> LockMapTransformer<K> keyMapped( 
128             final Function<K, @Nullable String> keyMapper ) {
129 
130         return ( k, p ) -> transformAcquire( keyMapper.apply( k ), p );
131 
132     }
133 
134     /**
135      * Derives a lock map transformer that does not add the lock key as a label in the
136      * observation.
137      *
138      * @param <K> The key type.
139      * @return The transformer.
140      * @see #transformAcquire(String, Mono)
141      */
142     @SideEffectFree
143     public <K extends @NonNull Object> LockMapTransformer<K> noKey() {
144 
145         return ( k, p ) -> transformAcquire( null, p );
146 
147     }
148 
149     @Override
150     public Mono<Void> transformAwait( final Mono<Void> pending ) {
151 
152         return pending.name( "latch.await" )
153                 .tag( "latch.name", name )
154                 .tap( Micrometer.observation( registry ) );
155 
156     }
157     
158 }
