3   import java.util.concurrent.atomic.AtomicBoolean;
4   
5   /**
6    * Base implementation for acquired locks.
7    *
8    * @version 1.0
9    * @since 1.0
10   */
11  abstract class AbstractAcquiredLock implements AcquiredLock {
12  
13      /** Whether the lock is currently active. */
14      private final AtomicBoolean active = new AtomicBoolean( true );
15  
16      /**
17       * Performs the actual release of the lock.
18       */
19      protected abstract void doRelease();
20  
21      @Override
22      public void release() {
23  
24          if ( active.getAndSet( false ) ) {
25              doRelease(); // Only release if currently active
26          }
27  
28      }
29      
30  }
