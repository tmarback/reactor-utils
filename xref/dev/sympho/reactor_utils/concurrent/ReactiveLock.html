1   package dev.sympho.reactor_utils.concurrent;
2   
3   import reactor.core.publisher.Flux;
4   import reactor.core.publisher.Mono;
5   
6   /**
7    * Lock that allows waiting for availability using reactive streams.
8    * 
9    * <p>As with its superinterface, this lock type is <b>NOT reentrant</b>. Waiting on a lock
10   * from a task that has already acquired it <b>WILL</b> cause a deadlock.
11   *
12   * @version 1.0
13   * @since 1.0
14   */
15  public interface ReactiveLock extends Lock {
16  
17      /**
18       * Acquires the lock.
19       * 
20       * <p>Note that each subscription to the returned Mono will be a <i>different</i> acquisition
21       * of the lock, allowing the Mono to be re-used. If acquiring the lock only once for all
22       * subscribers is desired, use {@link Mono#cache()} on the returned Mono, but do be careful
23       * to make sure that the lock is only released when all subscribers are done.
24       * 
25       * <p>It is safe to cancel a subscription made to the returned Mono. If a cancellation is
26       * detected, the lock will simply be re-released as soon as it is acquired. Do remember,
27       * however, that a downstream cancellation will not be propagated if this stage has already
28       * completed, in which case the lock must be released downstream.
29       *
30       * @return A Mono that issues the lock once acquired.
31       * @implSpec The current implementation operates in a FIFO manner. That is, when a lock is
32       *           released, it is given to the task that has been waiting for the longest amount
33       *           of time.
34       */
35      Mono<AcquiredLock> acquire();
36  
37      /**
38       * Guards a Mono with this lock, acquiring the lock before subscribing to the Mono
39       * and releasing the lock when it completes. This ensures that only one subscription
40       * is active at any given time.
41       *
42       * @param <T> The element type.
43       * @param source The Mono to guard.
44       * @return The guarded Mono.
45       */
46      default <T> Mono<T> guard( final Mono<T> source ) {
47  
48          return Mono.usingWhen( 
49                  acquire(), // Acquire lock
50                  l -> source, // Relay mono
51                  l -> Mono.fromRunnable( l::release ) // Release lock
52          );
53  
54      }
55  
56      /**
57       * Guards a Flux with this lock, acquiring the lock before subscribing to the Flux
58       * and releasing the lock when it completes. This ensures that only one subscription
59       * is active at any given time.
60       *
61       * @param <T> The element type.
62       * @param source The Flux to guard.
63       * @return The guarded Flux.
64       */
65      default <T> Flux<T> guard( final Flux<T> source ) {
66  
67          return Flux.usingWhen( 
68                  acquire(), // Acquire lock
69                  l -> source, // Relay flux
70                  l -> Mono.fromRunnable( l::release ) // Release lock
71          );
72  
73      }
74  
75  }
